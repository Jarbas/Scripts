
SELECT OPE.NUMERO, OPE.COD_ETIQUETA,OPE.ETIQUETA_GERADA, PE.SEQUENCIA, OP.NUMERO_OP, PE.* FROM PRODUTOS_ETIQUETAS PE
LEFT JOIN ORDENS_PRODUCAO_ETIQUETAS OPE ON PE.COD_ETIQUETA = OPE.ETIQUETA_GERADA 
LEFT JOIN ORDENS_PRODUCAO OP ON PE.COD_ORDEM_PRODUCAO = OP.COD_OP  
WHERE PE.DATA_GERACAO > TO_DATE ('01/01/2014') AND PE.COD_TIPO_ETIQUETA = 1 AND OPE.NUMERO IS NULL AND PE.QUANTIDADE = 500 AND PE.STATUS = 'A';

SELECT  PE.SEQUENCIA,  OPE.COD_ETIQUETA, PE.COD_ETIQUETA, OPE.NUMERO, PE.* FROM PRODUTOS_ETIQUETAS  PE
JOIN ORDENS_PRODUCAO_ETIQUETAS OPE ON PE.COD_ETIQUETA =  OPE.ETIQUETA_GERADA
WHERE PE.DATA_GERACAO > TO_DATE ('01/01/2014')  AND PE.COD_TIPO_ETIQUETA = 1  AND ( PE.SEQUENCIA <> OPE.COD_ETIQUETA );


SELECT  PE.SEQUENCIA,  PEL.COD_ETIQUETA, PE.COD_ETIQUETA, PEL.NUMERO_FARDO , PE.* FROM PRODUTOS_ETIQUETAS  PE
JOIN PLANO_EMBARQUE_LEITURA  PEL ON PE.COD_ETIQUETA =  PEL.COD_ETIQUETA
WHERE PE.DATA_GERACAO > TO_DATE ('01/01/2014')  AND PE.COD_TIPO_ETIQUETA = 1  AND ( PE.SEQUENCIA <> PEL.NUMERO_FARDO );
