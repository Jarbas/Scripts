SELECT PE.COD_PRODUTO, PR.DESCRICAO, PA.COD_UNIDADE_NEGOCIO, OP.NUMERO_OP,  PE.STATUS , PE.DATA_GERACAO, PE.STATUS, PE.PESO, PE.QUANTIDADE, PE.TURNO FROM PRODUTOS_ETIQUETAS PE
LEFT JOIN  PRODUTOS PR ON PE.COD_PRODUTO = PR.COD_PRODUTO
LEFT JOIN PRODUTOS_ACABADOS PA ON PE.COD_PRODUTO = PA.COD_PRODUTO
LEFT JOIN ORDENS_PRODUCAO OP ON PE.COD_ORDEM_PRODUCAO = OP.COD_OP
WHERE PE.COD_TIPO_ETIQUETA IN (1,2) AND PE.STATUS <> 'C'
AND TRUNC(PE.DATA_GERACAO) BETWEEN TO_DATE(:dat1) AND TO_DATE('06/02/2015')
AND NOT PR.DESCRICAO LIKE '%SEGUNDA%' AND NOT PA.COD_UNIDADE_NEGOCIO IN (2,4)
ORDER BY PE.DATA_GERACAO;


SELECT PE.COD_PRODUTO, PR.DESCRICAO, PA.COD_UNIDADE_NEGOCIO, OP.NUMERO_OP,  PE.STATUS , PE.DATA_GERACAO, PE.STATUS, PE.PESO, PE.QUANTIDADE, PE.TURNO FROM PRODUTOS_ETIQUETAS PE
LEFT JOIN  PRODUTOS PR ON PE.COD_PRODUTO = PR.COD_PRODUTO
LEFT JOIN PRODUTOS_ACABADOS PA ON PE.COD_PRODUTO = PA.COD_PRODUTO
LEFT JOIN ORDENS_PRODUCAO OP ON PE.COD_ORDEM_PRODUCAO = OP.COD_OP
WHERE PE.COD_TIPO_ETIQUETA IN (1,2) AND PE.STATUS <> 'C'
AND TRUNC(PE.DATA_GERACAO) BETWEEN TO_DATE('01/02/2015') AND TO_DATE('06/02/2015')
AND  PR.DESCRICAO LIKE '%SEGUNDA%' 
ORDER BY PE.DATA_GERACAO;
